Private Function CNameToString(ByVal cdata As LMSTestLabAutomation.IData) As String
    Dim content As String
    If cdata Is Nothing Then
      content = ""
    Else
      content = cdata.Item("Contents")
      If content Is Nothing Then
        content = ""
      Else
        Dim length As Long
        length = Len(content)
        content = Left(content, length - 1)
        content = Right(content, length - 2)
      End If
    End If
    CNameToString = content
End Function

Private Function PathToRunName(ByVal path As String) As String
  On Error GoTo ErrLine
  Dim run, delim As String
  Dim index, length As Long
  delim = "***!DELIMITER!***"
  run = Replace(path, "\/", delim)
  index = InStr(1, run, "/")
  length = Len(run)
  run = Right(run, length - index)
  index = InStr(1, run, "/")
  If Not index = 0 Then run = Left(run, index)
  PathToRunName = Replace(run, delim, "\/")
  Exit Function
ErrLine:
  PathToRunName = ""
End Function

Private Sub ButtonFilter_Click()
  Set filteredRuns = New Collection
  Dim chkbox As Variant
  For Each chkbox In Frame1.Controls
    If chkbox.Value Then Call filteredRuns.Add(chkbox.Caption)
  Next
  
  If filteredRuns.Count = 0 Then
    Call MsgBox("No runs are selected!", vbExclamation)
    Exit Sub
  End If
  
  Dim id As LMSTestLabAutomation.IData
  Dim run, path As String
  Dim paths As New Collection

  Dim i As Integer
  For i = 0 To ib.Count - 1
    Set id = ib.Item(i, True)
    path = id.Item("PathToTop")
    run = PathToRunName(path)
    If Not run = "" Then
      If ListContains(filteredRuns, run) Then Call paths.Add(path)
    End If
  Next

  Call ib.Clear
  On Error Resume Next
  For i = paths.Count To 1 Step -1
    Call DB.AddToInputBasket(paths(i))
  Next
  
  Call Frame1.Controls.Clear
  Dim offset As Integer
  offset = 6
  For i = 1 To filteredRuns.Count
    Set chkbox = Frame1.Controls.Add("Forms.CheckBox.1")
    chkbox.Caption = filteredRuns(i)
    chkbox.Left = 6
    chkbox.Top = offset
     '2 added lines
    chkbox.Width = 510
    chkbox.Font.Size = 14
    offset = offset + 18
  Next
  Call RefreshCheckedRuns
End Sub

Private Sub RefreshCheckedRuns()
  Dim i, iStart, iStep As Integer
  Dim chkbox As Variant
  For Each chkbox In Frame1.Controls
    chkbox.Value = False
  Next
  
  If IsNumeric(TextBox1.Text) Then iStart = Val(TextBox1.Text) Else iStart = 1
  If IsNumeric(TextBox2.Text) Then iStep = Val(TextBox2.Text) Else iStep = 1
  For i = iStart - 1 To Frame1.Controls.Count - 1 Step iStep
    Frame1.Controls.Item(i).Value = True
  Next
End Sub

Private Function ListContains(ByVal list As Collection, ByVal match As String) As Boolean
  Dim found As Boolean
  found = False
  Dim i As Integer
  For i = 1 To list.Count
    If list.Item(i) = match Then
      found = True
      Exit For
    End If
  Next
  ListContains = found
End Function

Private Function ListSort(ByVal list As Collection) As Collection
  If list.Count = 0 Then
    Set ListSort = list
    Exit Function
  End If
  
  Dim sorted As New Collection
  Call sorted.Add(list(1))
  
  Dim this, that As String
  Dim i, j As Integer
  Dim placed As Boolean
  For i = 2 To list.Count
    this = list(i)
    placed = False
    For j = 1 To i - 1
      that = sorted(j)
      If StrComp(this, that, 1) = -1 Then
        Call sorted.Add(this, before:=j)
        placed = True
        Exit For
      End If
      If Not placed Then Call sorted.Add(this)
    Next
  Next
  Set ListSort = sorted
End Function
